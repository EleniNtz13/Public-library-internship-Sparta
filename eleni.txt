from django.core.management.base import BaseCommand
import pandas as pd
from main.models import Book

# --- Συνάρτηση για ασφαλή μετατροπή ημερομηνίας ---
def parse_date(value):
    if pd.isna(value):
        return None
    try:
        # Αν η τιμή είναι ήδη datetime, παίρνουμε μόνο την ημερομηνία
        if isinstance(value, pd.Timestamp):
            return value.date()
        # Αν είναι string ή αριθμός, μετατρέπουμε σε datetime
        return pd.to_datetime(value).date()
    except Exception:
        return None

class Command(BaseCommand):
    help = "Import books data from Excel into PostgreSQL"

    def handle(self, *args, **options):
        # Διαβάζουμε το Excel
        df = pd.read_excel('main/excel_data/data.xlsx')

        # --- DEBUG: ελέγχουμε τις στήλες και τις πρώτες γραμμές ---
        print("Columns:", df.columns.tolist())
        print(df.head())

        # Δημιουργία Book objects
        books = []
        for _, row in df.iterrows():
            book = Book(
                entry_number=row.get('entry_number'),
                entry_date=parse_date(row.get('entry_date')),  # Χρήση της συνάρτησης εδώ
                author=row.get('author'),
                koha_author=row.get('koha_author'),
                title=row.get('title'),
                publisher=row.get('publisher'),
                edition=row.get('edition'),
                publish_year=row.get('publish_year'),
                publish_place=row.get('publish_place'),
                shape=row.get('shape'),
                pages=row.get('pages'),
                volume=row.get('volume'),
                notes=row.get('notes'),
                isbn=row.get('isbn'),
                column1=row.get('column1'),
                column2=row.get('column2')
            )
            books.append(book)

        # Bulk create στη βάση
        Book.objects.bulk_create(books)
        self.stdout.write(self.style.SUCCESS("✔ Successfully imported all books!"))
